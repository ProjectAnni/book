# 专辑分类

为了显示更加有序的专辑列表，元数据仓库对专辑进行了一定的分类。分类的标准和 `Anni` 约定中基本相同，但通过 `toml` 表示。

```toml
# category 为主分类，涵盖一系列专辑
[category]
# 分类名称
name = "THE IDOLM@STER Shiny Colors"
# 分类类型
type = "AAAA"
# 专辑列表，填写该分类下所有专辑的 Catalog
# 为节省篇幅，在该文档中此处留空
albums = []

# inner 中存储主分类下各个子分类
[[category.inner]]
# 子分类名称
name = "シャイニーカラーズ"
# 子分类中各专辑
albums = [
  "LACM-14781",
  "LACM-14861",
  "LACM-14982",
  "LACM-14906",
  "LACM-14965"
]

[[category.inner]]
name = "イルミネーションスターズ"
albums = [
  "LACM-14782",
  "LACM-14862",
  "LACM-14983"
]

[[category.inner]]
name = "アンティーカ"
albums = [
  "LACM-14783",

  "LACM-14984"
]

[[category.inner]]

albums = [
  "LACM-14784",

  "LACM-14985"
]

[[category.inner]]
name = "アルストロメリア"
albums = [

  "LACM-14986"
]

[[category.inner]]

  "LACM-14866",

]

[[category.inner]]

  "LACM-24005"
]
```
